from flask import Blueprint, render_template, request, jsonify from flask_login import login_required from sqlalchemy import or_ from models import Ministro, Missa, Aviso, Escala, db busca_bp = Blueprint("busca", __name__) @busca_bp.route("/busca") @login_required def busca(): termo = (request.args.get("q") or "").strip() ministros = [] missas = [] avisos = [] escalas = [] if termo: # MINISTROS ministros = Ministro.query.filter( Ministro.nome.ilike(f"%{termo}%") ).order_by(Ministro.nome.asc()).all() # MISSAS missas = Missa.query.filter( or_( Missa.comunidade.ilike(f"%{termo}%"), Missa.horario.ilike(f"%{termo}%"), Missa.data.cast(db.String).ilike(f"%{termo}%") ) ).order_by(Missa.data.desc()).all() # AVISOS avisos = Aviso.query.filter( Aviso.titulo.ilike(f"%{termo}%") ).order_by(Aviso.id.desc()).all() # MISSAS ESCALADAS escalas = db.session.query(Escala, Ministro, Missa)\ .join(Ministro)\ .join(Missa)\ .filter( Ministro.nome.ilike(f"%{termo}%") )\ .order_by(Missa.data.desc())\ .all() return render_template( "busca.html", termo=termo, ministros=ministros, missas=missas, avisos=avisos, escalas=escalas )
